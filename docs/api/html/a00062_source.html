171  template<typename OtherDerived, typename ContractionDims>
172  TensorReference executeContraction(OtherDerived& t2,
173  ContractionDims& cIndices) {
174 
175  // Compute new Tensor rank
176  static constexpr int newRank = Rank + OtherDerived::getRank()
177  - 2 * array_size<ContractionDims>::value;
178 
179  // Create an Eigen Tensor from OtherDerived...
180  auto otherRef = t2.createReference();
181  Eigen::TensorMap<Eigen::Tensor<Scalar, OtherDerived::getRank()>> otherTensor =
182  getEigenTensorFromReference(otherRef);
183 
184  // ---------------------------------------------------------
185  // Need to compute dimensions of new tensor
186  Eigen::DSizes<Eigen::DenseIndex, newRank> newDimensions;
187  int counter = 0;
188  if (array_size<ContractionDims>::value != 0) {
189  for (int i = 0; i < tensor->dimensions().size(); i++) {
190  for (auto c : cIndices) {
191  if (i != c.first) {
192  newDimensions[counter] = tensor->dimension(i);
193  counter++;
194  }
195  }
196  }
197  for (int i = 0; i < otherTensor.dimensions().size(); i++) {
198  for (auto c : cIndices) {
199  if (i != c.second) {
200  newDimensions[counter] = otherTensor.dimension(i);
201  counter++;
202  }
203  }
204  }
205  } else {
206  // If there are no contraction dims, then keep all
207  // dimensions
208  for (int i = 0; i < tensor->dimensions().size(); i++) {
209  newDimensions[counter] = tensor->dimension(i);
210  counter++;
211  }
212  for (int i = 0; i < otherTensor.dimensions().size(); i++) {
213  newDimensions[counter] = otherTensor.dimension(i);
214  counter++;
215  }
216  }
217  // ---------------------------------------------------------
218 
219  // Perform the Tensor Contraction multi-threaded
220  Eigen::Tensor<Scalar, newRank> result(newDimensions);
221  result.device(*device.get()) = tensor->contract(otherTensor, cIndices);
222 
223  // Create TensorShape
224  std::vector<int> dimensions(newRank);
225  for (int i = 0; i < newRank; i++) {
226  dimensions[i] = result.dimension(i);
227  }
228  TensorShape newShape(dimensions);
229 
230  // Create and return a reference to the new Tensor.
231  TensorReference newReference = fire::make_tensor_reference(
232  result.data(), newShape);
233 
234  return newReference;
235  }
