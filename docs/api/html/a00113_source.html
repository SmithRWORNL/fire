190  template<typename OtherDerived, typename ContractionDims>
191  TensorReference<Scalar> executeContraction(OtherDerived& t2,
192  ContractionDims& cIndices) {
193 
194  // Compute new Tensor rank
195  static constexpr int newRank = Rank + OtherDerived::getRank()
196  - 2 * array_size<ContractionDims>::value;
197 
198  // Create an Eigen Tensor from OtherDerived...
199  auto otherRef = t2.createReference();
200  Eigen::TensorMap<Eigen::Tensor<Scalar, OtherDerived::getRank()>> otherTensor =
201  getEigenTensorFromReferenceWithRank<OtherDerived::getRank()>(
202  otherRef);
203 
204  // ---------------------------------------------------------
205  // Need to compute dimensions of new tensor
206  Eigen::DSizes<Eigen::DenseIndex, newRank> newDimensions;
207  if (newRank != 0) {
208  int counter = 0;
209  if (array_size<ContractionDims>::value != 0) {
210  for (int i = 0; i < tensor->dimensions().size(); i++) {
211  for (auto c : cIndices) {
212  if (i != c.first) {
213  newDimensions[counter] = tensor->dimension(i);
214  counter++;
215  }
216  }
217  }
218  for (int i = 0; i < otherTensor.dimensions().size(); i++) {
219  for (auto c : cIndices) {
220  if (i != c.second) {
221  newDimensions[counter] = otherTensor.dimension(i);
222  counter++;
223  }
224  }
225  }
226  } else {
227  // If there are no contraction dims, then keep all
228  // dimensions
229  for (int i = 0; i < tensor->dimensions().size(); i++) {
230  newDimensions[counter] = tensor->dimension(i);
231  counter++;
232  }
233  for (int i = 0; i < otherTensor.dimensions().size(); i++) {
234  newDimensions[counter] = otherTensor.dimension(i);
235  counter++;
236  }
237  }
238  }
239 
240  // ---------------------------------------------------------
241 
242  // Perform the Tensor Contraction multi-threaded
243  Eigen::Tensor<Scalar, newRank> result(newDimensions);
244  result.device(*device.get()) = tensor->contract(otherTensor, cIndices);
245 
246  // Create TensorShape
247  std::vector<int> dimensions(newRank);
248  for (int i = 0; i < newRank; i++) {
249  dimensions[i] = result.dimension(i);
250  }
251  TensorShape newShape(dimensions);
252 
253  // Create and return a reference to the new Tensor.
254  TensorReference<Scalar> newReference = fire::make_tensor_reference(
255  result.data(), newShape);
256 
257  return newReference;
258  }

394  std::tuple<TensorReference<Scalar>, TensorReference<Scalar>, TensorReference<Scalar>> computeSvd(
395  TensorReference<Scalar>& ref, double cutoff) {
396 
397  // Get the data and shape
398  auto shape = ref.second;
399  auto data = ref.first.data();
400 
401  // Express the Tensor as an Eigen Matrix. We know
402  // at this point the Tensor is Rank 2. Double check anyway
403  assert(shape.dimensions().size() == 2);
404 
405  Eigen::Map<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>> matrix(data, shape.dimension(0),
406  shape.dimension(1));
407 
408  Eigen::BDCSVD<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>> svd(matrix,
409  Eigen::ComputeThinU | Eigen::ComputeThinV);
410 
411  auto singularValues = svd.singularValues();
412  auto u = svd.matrixU();
413  auto v = svd.matrixV();
414 
415  // These values are in decreasing order,
416  // so we just need to get the first one less than
417  // the cutoff, the rest will be too
418  int truncIndex = singularValues.rows() - 1;
419  for (int i = 0; i < singularValues.rows(); i++) {
420  if (singularValues(i) < cutoff) {
421  truncIndex = i;
422  break;
423  }
424  }
425 
426  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> S(truncIndex + 1, truncIndex + 1);
427  S.setZero();
428  for (int i = 0; i < truncIndex + 1; i++)
429  S(i, i) = singularValues(i);
430 
431  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> truncatedU = u.block(0, 0, truncIndex + 1,
432  truncIndex + 1);
433  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> truncatedV = v.block(0, 0, truncIndex + 1,
434  truncIndex + 1);
435 
436  assert ( (u*S*v.transpose() - matrix).array().abs().sum() < 1e-6 );
437 
438  std::vector<int> uDims(2), sDims(2), vDims(2);
439  uDims[0] = truncatedU.rows();
440  uDims[1] = truncatedU.cols();
441  sDims[0] = S.rows();
442  sDims[1] = S.cols();
443  vDims[0] = truncatedV.rows();
444  vDims[1] = truncatedV.cols();
445 
446  TensorShape uShape(uDims), sShape(sDims), vShape(vDims);
447 
448  // Shape should be the same as the given tensor ref
449  // for the computed U and V tensors
450  return std::make_tuple(
451  fire::make_tensor_reference(truncatedU.data(), uShape),
452  fire::make_tensor_reference(S.data(), sShape),
453  fire::make_tensor_reference(truncatedV.data(), vShape));
454 
455  }
