276  void read_request_and_content(const std::shared_ptr<socket_type> &socket) {
277  //Create new streambuf (Request::streambuf) for async_read_until()
278  //shared_ptr is used to pass temporary objects to the asynchronous functions
279  std::shared_ptr<Request> request(new Request(*socket));

284  boost::asio::async_read_until(*socket, request->streambuf, "\r\n\r\n",
285  [this, socket, request, timer](const boost::system::error_code& ec, size_t bytes_transferred) {
286  if(timer)
287  timer->cancel();
288  if(!ec) {
289  //request->streambuf.size() is not necessarily the same as bytes_transferred, from Boost-docs:
290  //"After a successful async_read_until operation, the streambuf may contain additional data beyond the delimiter"
291  //The chosen solution is to extract lines from the stream directly when parsing the header. What is left of the
292  //streambuf (maybe some bytes of the content) is appended to in the async_read-function below (for retrieving content).

336  bool parse_request(const std::shared_ptr<Request> &request) const {
337  std::string line;
338  getline(request->content, line);
339  size_t method_end;
340  if((method_end=line.find(' '))!=std::string::npos) {
341  size_t path_end;
342  if((path_end=line.find(' ', method_end+1))!=std::string::npos) {
343  request->method=line.substr(0, method_end);
344  request->path=line.substr(method_end+1, path_end-method_end-1);
345 
346  size_t protocol_end;
347  if((protocol_end=line.find('/', path_end+1))!=std::string::npos) {
348  if(line.compare(path_end+1, protocol_end-path_end-1, "HTTP")!=0)
349  return false;
350  request->http_version=line.substr(protocol_end+1, line.size()-protocol_end-2);
351  }
352  else
353  return false;
354 
355  getline(request->content, line);
356  size_t param_end;
357  while((param_end=line.find(':'))!=std::string::npos) {
358  size_t value_start=param_end+1;
359  if((value_start)<line.size()) {
360  if(line[value_start]==' ')
361  value_start++;
362  if(value_start<line.size())
363  request->header.emplace(line.substr(0, param_end), line.substr(value_start, line.size()-value_start-1));
364  }
365 
366  getline(request->content, line);
367  }
368  }
369  else
370  return false;
371  }
372  else
373  return false;
374  return true;
375  }
