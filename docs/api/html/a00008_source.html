232  void buildFluxMaps() {
233  // These tempInt blocks will become fPlusMap and fMinusMap eventually.
234  const size_t tempIntSize = numSpecies * numReactions / 2;
235  vector<int> tempInt1(tempIntSize,0);
236  vector<int> tempInt2(tempIntSize,0);
237 
238  // Access elements by reacMask[speciesIndex + species * reactionIndex].
239  vector<int> reacMask(numSpecies * numReactions,0); // [species][reactions]
240  vector<int> numFluxPlus(numSpecies,0), numFluxMinus(numSpecies,0);
241 
242  // Start of Guidry's original parseF() code
243  int incrementPlus = 0;
244  int incrementMinus = 0;
245  int localNumFPlus = 0, localNumFMinus = 0;
246  numFPlus = 0; numFMinus = 0;
247 
248  // Loop over all isotopes in the network
249 
250  for (int i = 0; i < numSpecies; i++) {
251  int total = 0;
252  localNumFPlus = 0;
253  localNumFMinus = 0;
254 
255  // Loop over all possible reactions for this isotope, finding those that
256  // change its population up (contributing to F+) or down (contributing
257  // to F-).
258  Species currentSpecies = species->at(i);
259  for (int j = 0; j < numReactions; j++) {
260  int totalL = 0;
261  int totalR = 0;
262 
263  // Loop over reactants for this reaction
264  Reaction & reaction = reactions->at(j);
265  for (int k = 0; k < reaction.numReactants; k++) {
266  if (currentSpecies.atomicNumber == reaction.reactantZ[k]
267  && currentSpecies.neutronNumber
268  == reaction.reactantN[k])
269  totalL++;
270  }
271 
272  // Loop over products for this reaction
273  for (int k = 0; k < reaction.numProducts; k++) {
274  if (currentSpecies.atomicNumber == reaction.productZ[k]
275  && currentSpecies.neutronNumber
276  == reaction.productN[k])
277  totalR++;
278  }
279 
280  total = totalL - totalR;
281 
282  if (total > 0) // Contributes to F- for this isotope
283  {
284  localNumFMinus++;
285  reacMask[i + numSpecies * j] = -total;
286  tempInt2[incrementMinus + localNumFMinus - 1] = j;
287  } else if (total < 0) // Contributes to F+ for this isotope
288  {
289  localNumFPlus++;
290  reacMask[i + numSpecies * j] = -total;
291  tempInt1[incrementPlus + localNumFPlus - 1] = j;
292  } else // Does not contribute to flux for this isotope
293  {
294  reacMask[i + numSpecies * j] = 0;
295  }
296  }
297 
298  // Keep track of the total number of F+ and F- terms in the network for all isotopes
299  numFPlus += localNumFPlus;
300  numFMinus += localNumFMinus;
301 
302  numFluxPlus[i] = localNumFPlus;
303  numFluxMinus[i] = localNumFMinus;
304 
305  incrementPlus += localNumFPlus;
306  incrementMinus += localNumFMinus;
307 
308  // if (showParsing == 1)
309  // printf("%d %s numF+ = %d numF- = %d\n", i, isoLabel[i], numFplus, numFminus);
310  }
311 
312  // Create 1D arrays that will be used to map finite F+ and F- to the Flux array.
313  int FplusIsotopeCut[numSpecies];
314  int FminusIsotopeCut[numSpecies];
315  int FplusIsotopeIndex[numFPlus];
316  int FminusIsotopeIndex[numFMinus];
317 
318  FplusIsotopeCut[0] = numFluxPlus[0];
319  FminusIsotopeCut[0] = numFluxMinus[0];
320  for (int i = 1; i < numSpecies; i++) {
321  FplusIsotopeCut[i] = numFluxPlus[i] + FplusIsotopeCut[i - 1];
322  FminusIsotopeCut[i] = numFluxMinus[i] + FminusIsotopeCut[i - 1];
323  }
324 
325  int currentIso = 0;
326  for (int i = 0; i < numFPlus; i++) {
327  FplusIsotopeIndex[i] = currentIso;
328  if (i == (FplusIsotopeCut[currentIso] - 1))
329  currentIso++;
330  }
331 
332  currentIso = 0;
333  for (int i = 0; i < numFMinus; i++) {
334  FminusIsotopeIndex[i] = currentIso;
335  if (i == (FminusIsotopeCut[currentIso] - 1))
336  currentIso++;
337  }
338 
339  fPlusMap.resize(numFPlus);
340  for (int i = 0; i < numFPlus; i++) {
341  fPlusMap[i] = tempInt1[i];
342  }
343  fMinusMap.resize(numFMinus);
344  for (int i = 0; i < numFMinus; i++) {
345  fMinusMap[i] = tempInt2[i];
346  }
347 
348  // Populate the fPlusMin and FplusMax arrays
349  unsigned short fPlusMin [numSpecies];
350  unsigned short fMinusMin [numSpecies];
351 
352  fPlusMin[0] = 0;
353  fPlusMaximums.resize(numSpecies);
354  fPlusMaximums[0] = numFluxPlus[0] - 1;
355  for (int i = 1; i < numSpecies; i++) {
356  fPlusMin[i] = fPlusMaximums[i - 1] + 1;
357  fPlusMaximums[i] = fPlusMin[i] + numFluxPlus[i] - 1;
358  }
359  // Populate the fMinusMin and FminusMax arrays
360  fMinusMin[0] = 0;
361  fMinusMaximums.resize(numSpecies);
362  fMinusMaximums[0] = numFluxMinus[0] - 1;
363  for (int i = 1; i < numSpecies; i++) {
364  fMinusMin[i] = fMinusMaximums[i - 1] + 1;
365  fMinusMaximums[i] = fMinusMin[i] + numFluxMinus[i] - 1;
366  }
367 
368  // Allocate the flux vectors
369  fPlusFactors.resize(numSpecies*numReactions);
370  fMinusFactors.resize(numSpecies*numReactions);
371 
372  // Populate the fPlusFac and FminusFac arrays that hold the factors counting the
373  // number of occurences of the species in the reaction. Note that this can only
374  // be done after parseF() has been run to give reacMask[i][j].
375  int tempCountPlus = 0;
376  int tempCountMinus = 0;
377  for (int i = 0; i < numSpecies; i++) {
378  for (int j = 0; j < numReactions; j++) {
379  if (reacMask[i + numSpecies * j] > 0) {
380  fPlusFactors[tempCountPlus] = reacMask[i + numSpecies * j];
381  tempCountPlus++;
382  } else if (reacMask[i + numSpecies * j] < 0) {
383  fMinusFactors[tempCountMinus] = -1.0*reacMask[i
384  + numSpecies * j];
385  tempCountMinus++;
386  }
387  }
388  } // If you run this through the debugger, it is a lot of work to calculate 0!
389 
390  // Notice that fPlusFactors == fMinusFactors! ~ Question for Mike!
391 
392  return;
393  }

414  void computeRates(const double & temp) {
415  // Compute the temperatures
416  array<double,6> tempValues;
417  double cbrtT = cbrt(temp); // Cube root of T
418  tempValues[0] = 1 / temp; // 1/T
419  tempValues[1] = 1 / cbrtT;
420  tempValues[2] = cbrtT;
421  tempValues[3] = temp;
422  tempValues[4] = cbrtT * cbrtT * cbrtT * cbrtT * cbrtT; // T^(5/3)
423  tempValues[5] = log(temp);
424 
425  // Compute the rates
426  for (Reaction & reaction : *reactions) {
427  reaction.setRate(tempValues);
428  }
429  }

436  void computeFluxes() {
437 
438  vector<double> reactionFlux(numReactions, 0);
439 
440  // Compute the flux due to each reaction
441  for (int i = 0; i < numReactions; i++) {
442  Reaction & reaction = reactions->at(i);
443  // This computes the flux as the product of the rate times some
444  // power of the current abundance. It is written as presented below
445  // to (hopefully) streamline the performance. There is a branchless
446  // implementation too, but it isn't clear to me that it is faster.
447  // 1 Body
448  reactionFlux[i] = reaction.rate
449  * (species->at(reaction.reactants[0]).massFraction);
450  // 2 Body
451  if (reaction.numReactants > 1)
452  reactionFlux[i] *=
453  (species->at(reaction.reactants[1]).massFraction);
454  // 3 Body
455  else if (reaction.numReactants > 2)
456  reactionFlux[i] *=
457  (species->at(reaction.reactants[1]).massFraction)
458  * (species->at(reaction.reactants[2]).massFraction);
459  }
460 
461  // Populate the incoming (fPlus) and outgoing flux (fMinus).
462  vector<double> fPlus(numFPlus, 0);
463  for (int i = 0; i < numFPlus; i++) {
464  fPlus[i] = fPlusFactors[i] * reactionFlux[fPlusMap[i]];
465  }
466  vector<double> fMinus(numFMinus, 0);
467  for (int i = 0; i < numFMinus; i++) {
468  fMinus[i] = fMinusFactors[i] * reactionFlux[fMinusMap[i]];
469  }
470 
471  // Sum the F+ and F- for each isotope. These are "sub-arrays"
472  // of Fplus and Fminus at (F[+ or -] + minny) of size fPlusMaximums[i].
473  int minny;
474  vector<double> fPlusSum(numSpecies, 0);
475  vector<double> fMinusSum(numSpecies, 0);
476  for (int i = 0; i < numSpecies; i++) {
477  minny = (i > 0) ? fPlusMaximums[i - 1] + 1 : 0;
478  // Sum F+.
479  fPlusSum[i] = 0.0;
480  for (int j = minny; j <= fPlusMaximums[i]; j++) {
481  fPlusSum[i] += fPlus[j];
482  }
483  // Sum F-.
484  minny = (i > 0) ? fMinusMaximums[i - 1] + 1 : 0;
485  fMinusSum[i] = 0.0;
486  for (int j = minny; j <= fMinusMaximums[i]; j++) {
487  fMinusSum[i] += fMinus[j];
488  }
489  species->at(i).flux = fPlusSum[i] - fMinusSum[i];
490  }
491 
492  return;
493  }
