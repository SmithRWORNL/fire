368 
369  // Begin CVODE code. Adapted from their examples.
370  realtype dx, dy, reltol, abstol, tout, umax;
371  N_Vector u;
372  void *cvode_mem;
373  int iout, flag;
374  long int nst;
375 
376  u = NULL;
377  cvode_mem = NULL;
378  int size = state.size();
379 
380  /* Create a serial vector */
381  u = N_VNew_Serial(size);
382  /* Allocate the U vector */
383  if (fire::cvode::check_flag((void*) u, "N_VNew_Serial", 0))
384  return;
385 
386  reltol = ZERO; /* Set the tolerances */
387  abstol = ATOL;
388 
389  // Set the initial conditions in u from the states
390  fire::cvode::setICs<T>(u, state);
391 
392  /* Call CVodeCreate to create the solver memory and specify the
393  * Backward Differentiation Formula and the use of a Newton iteration */
394  cvode_mem = CVodeCreate(CV_BDF, CV_NEWTON);
395  if (fire::cvode::check_flag((void *) cvode_mem, "CVodeCreate", 0))
396  return;
397 
398  /* Call CVodeInit to initialize the integrator memory and specify the
399  * user's right hand side function in u'=f(t,u), the inital time T0, and
400  * the initial dependent variable vector u. */
401  flag = CVodeInit(cvode_mem, fire::cvode::f<T>, initialT, u);
402  if (fire::cvode::check_flag(&flag, "CVodeInit", 1))
403  return;
404 
405  /* Call CVodeSStolerances to specify the scalar relative tolerance
406  * and scalar absolute tolerance */
407  flag = CVodeSStolerances(cvode_mem, reltol, abstol);
408  if (fire::cvode::check_flag(&flag, "CVodeSStolerances", 1))
409  return;
410 
411  /* Set the pointer to user-defined data */
412  flag = CVodeSetUserData(cvode_mem, &state);
413  if (fire::cvode::check_flag(&flag, "CVodeSetUserData", 1))
414  return;
415 
416  /* Call CVBand to specify the CVBAND band linear solver */
417  flag = CVDense(cvode_mem, size);
418  if (fire::cvode::check_flag(&flag, "CVDense", 1))
419  return;
420 
421  /* Set the user-supplied Jacobian routine Jac */
422 // flag = CVDlsSetBandJacFn(cvode_mem, Jac);
423 // if(fire::cvode::check_flag(&flag, "CVDlsSetBandJacFn", 1)) return;
424 
425  /* In loop over output points: call CVode, print results, test for errors */
426  umax = N_VMaxNorm(u);
427  fire::cvode::PrintHeader(reltol, abstol, umax, initialT);
428  for (iout = 1, tout = DTOUT; iout <= maxNumOutputSteps && currentT < finalT; iout++, tout += DTOUT) {
429  flag = CVode(cvode_mem, tout, u, &currentT, CV_NORMAL);
430  if (fire::cvode::check_flag(&flag, "CVode", 1))
431  break;
432  umax = N_VMaxNorm(u);
433  flag = CVodeGetNumSteps(cvode_mem, &nst);
434  fire::cvode::check_flag(&flag, "CVodeGetNumSteps", 1);
435  // Update the state
436  realtype * udata = NV_DATA_S(u);
437  state.t(currentT);
438  state.u(udata);
439  // Replace! - Notify observers that the state has changed
440  fire::cvode::PrintOutput(currentT, umax, nst);
441  }
442 
443  fire::cvode::PrintFinalStats(cvode_mem); /* Print some final statistics */
444 
445  N_VDestroy_Serial(u); /* Free the u vector */
446  CVodeFree(&cvode_mem); /* Free the integrator memory */
447 
448  return;
449  }
