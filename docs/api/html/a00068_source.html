216  void buildFluxMaps() {
217  // These tempInt blocks will become fPlusMap and fMinusMap eventually.
218  const size_t tempIntSize = numSpecies * numReactions / 2;
219  unsigned short tempInt1 [tempIntSize];
220  unsigned short tempInt2 [tempIntSize];
221 
222  // Access elements by reacMask[speciesIndex + species * reactionIndex].
223  int reacMask[numSpecies * numReactions]; // [species][reactions]
224  int numFluxPlus [numSpecies];
225  int numFluxMinus[numSpecies];
226 
227  // Start of Guidry's original parseF() code
228  int incrementPlus = 0;
229  int incrementMinus = 0;
230 
231  int totalFplus = 0;
232  int totalFminus = 0;
233 
234  // Loop over all isotopes in the network
235 
236  for (int i = 0; i < numSpecies; i++) {
237  int total = 0;
238  int numFplus = 0;
239  int numFminus = 0;
240 
241  // Loop over all possible reactions for this isotope, finding those that
242  // change its population up (contributing to F+) or down (contributing
243  // to F-).
244  Species currentSpecies = species->at(i);
245  for (int j = 0; j < numReactions; j++) {
246  int totalL = 0;
247  int totalR = 0;
248 
249  // Loop over reactants for this reaction
250  Reaction & reaction = reactions->at(j);
251  for (int k = 0; k < reaction.numReactants; k++) {
252  if (currentSpecies.atomicNumber == reaction.reactantZ[k]
253  && currentSpecies.neutronNumber
254  == reaction.reactantN[k])
255  totalL++;
256  }
257 
258  // Loop over products for this reaction
259  for (int k = 0; k < reaction.numProducts; k++) {
260  if (currentSpecies.atomicNumber == reaction.productZ[k]
261  && currentSpecies.neutronNumber
262  == reaction.productN[k])
263  totalR++;
264  }
265 
266  total = totalL - totalR;
267 
268  if (total > 0) // Contributes to F- for this isotope
269  {
270  numFminus++;
271  reacMask[i + numSpecies * j] = -total;
272  tempInt2[incrementMinus + numFminus - 1] = j;
273  } else if (total < 0) // Contributes to F+ for this isotope
274  {
275  numFplus++;
276  reacMask[i + numSpecies * j] = -total;
277  tempInt1[incrementPlus + numFplus - 1] = j;
278  } else // Does not contribute to flux for this isotope
279  {
280  reacMask[i + numSpecies * j] = 0;
281  }
282  }
283 
284  // Keep track of the total number of F+ and F- terms in the network for all isotopes
285  totalFplus += numFplus;
286  totalFminus += numFminus;
287 
288  numFluxPlus[i] = numFplus;
289  numFluxMinus[i] = numFminus;
290 
291  incrementPlus += numFplus;
292  incrementMinus += numFminus;
293 
294  // if (showParsing == 1)
295  // printf("%d %s numF+ = %d numF- = %d\n", i, isoLabel[i], numFplus, numFminus);
296  }
297 
298  // Create 1D arrays that will be used to map finite F+ and F- to the Flux array.
299  int FplusIsotopeCut[numSpecies];
300  int FminusIsotopeCut[numSpecies];
301  int FplusIsotopeIndex[totalFplus];
302  int FminusIsotopeIndex[totalFminus];
303 
304  FplusIsotopeCut[0] = numFluxPlus[0];
305  FminusIsotopeCut[0] = numFluxMinus[0];
306  for (int i = 1; i < numSpecies; i++) {
307  FplusIsotopeCut[i] = numFluxPlus[i] + FplusIsotopeCut[i - 1];
308  FminusIsotopeCut[i] = numFluxMinus[i] + FminusIsotopeCut[i - 1];
309  }
310 
311  int currentIso = 0;
312  for (int i = 0; i < totalFplus; i++) {
313  FplusIsotopeIndex[i] = currentIso;
314  if (i == (FplusIsotopeCut[currentIso] - 1))
315  currentIso++;
316  }
317 
318  currentIso = 0;
319  for (int i = 0; i < totalFminus; i++) {
320  FminusIsotopeIndex[i] = currentIso;
321  if (i == (FminusIsotopeCut[currentIso] - 1))
322  currentIso++;
323  }
324 
325  fPlusMap.resize(totalFplus);
326  for (int i = 0; i < totalFplus; i++) {
327  fPlusMap[i] = tempInt1[i];
328  }
329  fMinusMap.resize(totalFminus);
330  for (int i = 0; i < totalFminus; i++) {
331  fMinusMap[i] = tempInt2[i];
332  }
333 
334  // Populate the FplusMin and FplusMax arrays
335  unsigned short FplusMin [numSpecies];
336  unsigned short FminusMin [numSpecies];
337 
338  FplusMin[0] = 0;
339  fPlusMaximums.resize(numSpecies);
340  fPlusMaximums[0] = numFluxPlus[0] - 1;
341  for (int i = 1; i < numSpecies; i++) {
342  FplusMin[i] = fPlusMaximums[i - 1] + 1;
343  fPlusMaximums[i] = FplusMin[i] + numFluxPlus[i] - 1;
344  }
345  // Populate the FminusMin and FminusMax arrays
346  FminusMin[0] = 0;
347  fMinusMaximums.resize(numSpecies);
348  fMinusMaximums[0] = numFluxMinus[0] - 1;
349  for (int i = 1; i < numSpecies; i++) {
350  FminusMin[i] = fMinusMaximums[i - 1] + 1;
351  fMinusMaximums[i] = FminusMin[i] + numFluxMinus[i] - 1;
352  }
353 
354  // Allocate the flux vectors
355  fPlusFactors.resize(numSpecies*numReactions);
356  fMinusFactors.resize(numSpecies*numReactions);
357 
358  // Populate the fPlusFac and FminusFac arrays that hold the factors counting the
359  // number of occurences of the species in the reaction. Note that this can only
360  // be done after parseF() has been run to give reacMask[i][j].
361  int tempCountPlus = 0;
362  int tempCountMinus = 0;
363  for (int i = 0; i < numSpecies; i++) {
364  for (int j = 0; j < numReactions; j++) {
365  if (reacMask[i + numSpecies * j] > 0) {
366  fPlusFactors[tempCountPlus] = reacMask[i + numSpecies * j];
367  tempCountPlus++;
368  } else if (reacMask[i + numSpecies * j] < 0) {
369  fMinusFactors[tempCountMinus] = -1.0*reacMask[i
370  + numSpecies * j];
371  tempCountMinus++;
372  }
373  }
374  } // If you run this through the debugger, it is a lot of work to calculate 0!
375

377 
378  return;
379  }

396  void computeRates(const double & temp) {
397  // Compute the temperatures
398  array<double,6> tempValues;
399  double cbrtT = cbrt(temp); // Cube root of T
400  tempValues[0] = 1 / temp; // 1/T
401  tempValues[1] = 1 / cbrtT;
402  tempValues[2] = cbrtT;
403  tempValues[3] = temp;
404  tempValues[4] = cbrtT * cbrtT * cbrtT * cbrtT * cbrtT; // T^(5/3)
405  tempValues[5] = log(temp);
406 
407  // Compute the rates
408  for (Reaction & reaction : *reactions) {
409  reaction.setRate(tempValues);
410  }
411  }
